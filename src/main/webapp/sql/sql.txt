create table replyboard(
bid number(4) PRIMARY key,
bname varchar2(20),
btitle varchar2(100),
bcontent varchar2(300),
bdate Date default sysdate,

bgroup number(4), --원글의 bid
bstep number(4),--답글이 달린글들의 순서
bindent number(4));--들여쓰기

create sequence replyboard_seq;
